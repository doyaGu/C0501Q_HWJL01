76 static inline void ithDmaSetRequest(int ch, ITHDmaMode srcMode, ITHDmaRequest srcReq, ITHDmaMode dstMode, ITHDmaRequest dstReq)
77 {
78  uint32_t cfg = dmaCtxt[ch].priority | ITH_DMA_CFG_GW(GRANT_WINDOW);
79 
80  cfg |= (ITH_DMA_CFG_DST_HANDSHAKE_EN(dstMode) |
81  ITH_DMA_CFG_DST_HANDSHAKE(ch) |
82  ITH_DMA_CFG_SRC_HANDSHAKE_EN(srcMode) |
83  ITH_DMA_CFG_SRC_HANDSHAKE(ch));
84 
85 #if !defined(DMA_IRQ_ENABLE)
86  cfg |= ITH_DMA_INT_MASK;
87 #endif
88  ithWriteRegA((ITH_DMA_BASE + ITH_DMA_CFG_CH(ch)), cfg);
89 
90  if (srcMode == ITH_DMA_HW_HANDSHAKE_MODE) {
91  ithWriteRegMaskA((ITH_DMA_BASE + ITH_DMA_HW_SRC_SEL_S_REG(ch)),
92  ITH_DMA_HW_SRC_SEL_VAL(ch, srcReq),
93  ITH_DMA_HW_SRC_SEL_MSK(ch));
94  }
95 
96  if (dstMode == ITH_DMA_HW_HANDSHAKE_MODE) {
97  ithWriteRegMaskA((ITH_DMA_BASE + ITH_DMA_HW_SRC_SEL_D_REG(ch)),
98  ITH_DMA_HW_SRC_SEL_VAL(ch, dstReq),
99  ITH_DMA_HW_SRC_SEL_MSK(ch));
100  }
101 
102  dmaCtxt[ch].hw = srcMode | dstMode;
103  dmaCtxt[ch].srcHw = srcMode;
104  dmaCtxt[ch].dstHw = dstMode;
105 }
