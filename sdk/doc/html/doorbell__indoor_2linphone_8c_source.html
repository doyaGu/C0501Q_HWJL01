575 int LinphoneCheckDoNotDisturb(void)
576 {
577  time_t rawtime;
578  struct tm* timeinfo;
579  int currMins, startMins, endMins;
580 
581  if (!lpInited)
582  return 0;
583 
584  if (theConfig.do_not_disturb_mode != DONOT_DISTURB_TIME)
585  return 0;
586 
587  if (theConfig.do_not_disturb_start_hour == theConfig.do_not_disturb_end_hour &&
588  theConfig.do_not_disturb_start_min == theConfig.do_not_disturb_end_min)
589  return 0;
590 
591  time(&rawtime);
592  timeinfo = localtime(&rawtime);
593 
594  currMins = timeinfo->tm_hour * 60 + timeinfo->tm_min;
595  startMins = theConfig.do_not_disturb_start_hour * 60 + theConfig.do_not_disturb_start_min;
596  endMins = theConfig.do_not_disturb_end_hour * 60 + theConfig.do_not_disturb_end_min;
597 
598  if (lpInDoNotDisturb == false)
599  {
600  if (startMins < endMins)
601  {
602  if (currMins >= startMins && currMins < endMins)
603  {
604  LinphoneEnableDoNotDisturb();
605  return 1;
606  }
607  }
608  else
609  {
610  if (currMins >= startMins || currMins < endMins)
611  {
612  LinphoneEnableDoNotDisturb();
613  return 1;
614  }
615  }
616  }
617  else
618  {
619  if (startMins < endMins)
620  {
621  if (currMins < startMins || currMins >= endMins)
622  {
623  LinphoneDisableDoNotDisturb();
624  return -1;
625  }
626  }
627  else
628  {
629  if (currMins < startMins && currMins >= endMins)
630  {
631  LinphoneDisableDoNotDisturb();
632  return -1;
633  }
634  }
635  }
636  return 0;
637 }
