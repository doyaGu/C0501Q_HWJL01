64 static inline void ithDmaSetRequest(int ch, ITHDmaMode srcMode, ITHDmaRequest srcReq, ITHDmaMode dstMode, ITHDmaRequest dstReq)
65 {
66  uint32_t value;
67  value = (dstMode << ITH_DST_HE_BIT) |
68  (dstReq << ITH_DST_RS_BIT) |
69  (srcMode << ITH_SRC_HE_BIT) |
70  (srcReq << ITH_SRC_RS_BIT);
71 #if !defined(DMA_IRQ_ENABLE)
72  value |= ITH_DMA_INT_MASK;
73 #endif
74  ithWriteRegA((ITH_DMA_BASE+ITH_DMA_C0_CFG_REG+ch*ITH_DMA_CH_OFFSET), value);
75 
76  dmaCtxt[ch].hw = srcMode | dstMode;
77  dmaCtxt[ch].srcHw = srcMode;
